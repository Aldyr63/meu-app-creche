        // Função JavaScript para mostrar/esconder as respostas do FAQ
        function toggleAnswer(element) {
            const answer = element.nextElementSibling; // Pega o próximo elemento, que é a resposta
            element.classList.toggle('active'); // Adiciona/remove a classe 'active' para mudar o sinal +/-
            if (answer.classList.contains('show')) {
                answer.classList.remove('show');
            } else {
                answer.classList.add('show');
            }
        }
